```js:static/js/index.js
window.onload = function () {
  document.getElementById('my-button').addEventListener('click', function () {
    const counter = document.querySelector('#counter');
    const nowNum = Number(counter.innerHTML);
    counter.innerHTML = String(nowNum + 1);
  });
}
```

# SPA(シングルページアプリケーション)
## 導入1
バックエンドとフロントエンドの役割を理解する

```
app
 |--index.html
 |--api
 |   `--index.php
 `--static
     `--js
         `--index.js
```

```php:index.php
<?php
$str = [
    'data' => 'hello PHP'
];

header('Content-Type: application/json; charset=utf-8');
echo json_encode($str);
```

```html:index.html
<!doctype html>
<html>
  <head>
    <meta charset="utf-8">
  </head>
  <body>
    <div id="main"></div>
    <script src="static/js/index.js"></script>
  </body>
</html>
```

```js:index.js
fetch('http://localhost:8080/api')
  .then(function (res) {
    return res.json()
  })
  .then(function (res) {
    const main = document.getElementById('main')
    main.innerHTML = res.data
  })
```
## 導入2
```
app
 |--index.html
 |--route.php
 |--api
 |   `--index.php
 `--static
     `--js
         `--index.js
```
```bash
$ php -S localhost:8080 route.php
```
```route.php
<?php
$urlPath = explode('/', $_SERVER['REQUEST_URI']);
array_shift($urlPath);

if ($urlPath[0] === 'api') {
    require_once(__DIR__ . '/api/index.php');
} else if ($urlPath[0] === 'static') {
    return false;
} else {
    require_once(__DIR__ . '/index.html');
}
```
```html:index.html
<!doctype html>
<html>
  <head>
    <meta charset="utf-8">
  </head>
  <body>
    <header>
      <ul>
        <li><a href="/#/aaaa">aaaa</a></li>
        <li><a href="/#/bbbb">bbbb</a></li>
      </ul>
    </header>
    <main id="main"></main>
    <script src="static/js/index.js"></script>
  </body>
</html>
```
```js:index.js
const fetchData = function (params) {
  const baseUrl = 'http://localhost:8080/api/'
  fetch(baseUrl + params)
    .then(function (res) {
      return res.json()
    })
    .then(function (res) {
      const main = document.getElementById('main')
      main.innerHTML = res.data
    })
}

const parseUrl = function () {
  return location.hash.split('/')
}

let hash
hash = parseUrl()
fetchData(hash[1])

window.addEventListener('hashchange', function () {
  hash = parseUrl()
  fetchData(hash[1])
})
```
